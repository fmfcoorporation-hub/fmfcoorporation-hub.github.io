1. sum family

sum

deskripsi: menjumlahkan semua angka dalam rentang sel atau daftar argumen.

sintaks: =sum(number1, [number2], ...)

ab
110
220
330=sum(a1:a3)
hasil60

sumif

deskripsi: menjumlahkan angka dengan satu kriteria.

sintaks: =sumif(range, criteria, [sum_range])

ab
1apel100
2jeruk50
3apel150
4jeruk75
5=sumif(a1:a4, "apel", b1:b4)
hasil250

sumifs

deskripsi: menjumlahkan angka dengan banyak kriteria.

sintaks: =sumifs(sum_range, criteria_range1, criteria1, ...)

abc
1timurbuah100
2baratsayur50
3timursayur150
4baratbuah75
5=sumifs(c1:c4, a1:a4, "timur", b1:b4, "sayur")
hasil150

5=averageifs(c1:c4, a1:a4, "jakarta", b1:b4, ">30")
